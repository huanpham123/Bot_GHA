// particlesJS.load('particles', 'particles.json', function() { console.log('Particles loaded'); }); const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition; let recognition; if (SpeechRecognition) { recognition = new SpeechRecognition(); recognition.lang = 'vi-VN'; recognition.interimResults = false; recognition.maxAlternatives = 1; } else { alert("Trình duyệt không hỗ trợ nhận diện giọng nói."); } let isRecording = false; const recordButton = document.getElementById('record-button'); const chatBox = document.getElementById('chat-box'); const chatContent = document.getElementById('chat-content'); const loader = document.querySelector('.loader'); recordButton.addEventListener('click', toggleRecording); function toggleRecording() { if (!isRecording) { startRecognition(); } else { stopRecognition(); } } function startRecognition() { if (!recognition) return; isRecording = true; recordButton.classList.add('recording'); recognition.start(); } function stopRecognition() { if (!recognition) return; recognition.stop(); isRecording = false; recordButton.classList.remove('recording'); } recognition.onresult = function(event) { const transcript = event.results[0][0].transcript; console.log("Transcript:", transcript); sendChatMessage(transcript); }; recognition.onerror = function(event) { console.error("Speech recognition error:", event.error); alert("Lỗi nhận diện giọng nói: " + event.error); resetRecordButton(); }; recognition.onend = function() { recordButton.style.display = 'none'; chatBox.classList.add('active'); }; function sendChatMessage(message) { loader.style.display = 'block'; fetch('/chat', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ message: message }) }) .then(response => { if (!response.ok) throw new Error('Network response was not ok'); return response.json(); }) .then(data => { chatContent.innerText = data.final_answer || "Không nhận được phản hồi từ chatbot."; loader.style.display = 'none'; }) .catch(error => { console.error("Error sending message:", error); chatContent.innerText = "Đã xảy ra lỗi khi gửi tin nhắn."; loader.style.display = 'none'; }); } function resetRecordButton() { isRecording = false; recordButton.classList.remove('recording'); recordButton.innerHTML = ``; } document.getElementById('back-button').addEventListener('click', () => { chatBox.classList.remove('active'); setTimeout(() => { recordButton.style.display = 'flex'; resetRecordButton(); chatContent.innerText = ""; }, 500); }); document.addEventListener('mousemove', (e) => { const x = e.clientX / window.innerWidth; const y = e.clientY / window.innerHeight; document.body.style.background = `radial-gradient(at ${x * 100}% ${y * 100}%, #1a1a2e 0%, #0a0a16 50%)`; });
